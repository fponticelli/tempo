type DropN<N extends number, T extends any[]> = DropNImpl<N, T, []>

Head

type Head<T extends any[]> = T extends [infer H, ...any[]] ? H : never

Last

type Last<T extends any[]> = {
  n: Last<Tail<T>>
  one: T extends [infer H] ? H : never
  empty: never
}[T extends [] ? 'empty' : T extends [any] ? 'one' : 'n']

Length

type Length<T extends any[]> = T['length']

LoseLastImpl 

type LoseLastImpl<A extends any[], B extends any[]> = {
  empty: A
  next: LoseLastImpl<Prepend<B[0], A>, Tail<B>>
}[B extends [] ? 'empty' : B extends [any] ? 'empty' : 'next']

Prepend

type Prepend<Insert, Tail extends any[]> = ((
  i: Insert,
  ...tail: Tail
) => void) extends (...tuple: infer Tuple) => void
  ? Tuple
  : never

Reverse

type Reverse<T extends any[]> = ReverseImpl<T, [], []>

Tail

type Tail<T extends any[]> = ((...args: T) => void) extends (
  _: any,
  ...rest: infer Rest
) => void
  ? Rest
  : never

TupleToUnion 

type TupleToUnion<T extends any[]> = T[number]
