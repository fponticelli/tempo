ExcludeFunctionFields 

type ExcludeFunctionFields<T> = Pick<
  T,
  KeysWithoutFieldType<T, AnyFunction>
>

KeysWithFieldType 

type KeysWithFieldType<T, Condition> = {
  [K in keyof T]: WhenEquals<T[K], Condition, K>
}[keyof T]

KeysWithoutFieldType 

type KeysWithoutFieldType<T, Condition> = {
  [K in keyof T]: WhenEquals<T[K], Condition, never, K>
}[keyof T]

Merge

type Merge<A, B> = A & B

ObjectWithField 

type ObjectWithField<F extends IndexType, V> = { [_ in F]: V }

ObjectWithPath 

type ObjectWithPath<Path extends IndexType[], V> = Path extends []
  ? {}
  : Path extends [infer T]
  ? T extends IndexType
    ? { [_ in T]: V }
    : never
  : Path extends [infer K, ...any[]]
  ? K extends IndexType
    ? Tail<Path> extends infer Rest
      ? Rest extends IndexType[]
        ? { [_ in K]: ObjectWithPath<Rest, V> }
        : never
      : never
    : never
  : never

ReadonlyFields 

type ReadonlyFields<T> = Pick<T, ReadonlyKeys<T>>

ReadonlyKeys 

type ReadonlyKeys<T> = {
  [P in keyof T]-?: WhenEquals<
    { [Q in P]: T[P] },
    { -readonly [Q in P]: T[P] },
    never,
    P
  >
}[keyof T]

RemoveNullableFromFields 

type RemoveNullableFromFields<T> = {
  [K in keyof T]: NonNullable<T[K]>
}

TypeAtPath 

type TypeAtPath<Path extends IndexType[], O> = {
  next: Path extends [infer K, ...any[]]
    ? K extends IndexType
      ? Tail<Path> extends infer Rest
        ? O extends Record<IndexType, any>
          ? Rest extends IndexType[]
            ? TypeAtPath<Rest, O[K]>
            : never
          : never
        : never
      : never
    : never
  empty: O
  done: O extends Record<IndexType, any>
    ? Path extends [infer K]
      ? K extends IndexType
        ? O[K]
        : never
      : never
    : never
}[Path extends [] ? 'empty' : Path extends [any] ? 'done' : 'next']

WritableFields 

type WritableFields<T> = Pick<T, WritableKeys<T>>

WritableKeys 

type WritableKeys<T> = {
  [K in keyof T]-?: WhenEquals<
    { [Q in K]: T[K] },
    { -readonly [Q in K]: T[K] },
    K
  >
}[keyof T]
