module 'types/generic'

types

Cast

type Cast<A, B> = A extends B ? A : B

DeRef

type DeRef<T extends () => any> = T extends () => infer Ret ? Ret : never

Pointer

type Pointer<T> = () => T

WhenEquals 

type WhenEquals<X, Y, A = X, B = never> = Equals<X, Y> extends true ? A : B

WhenNotEquals 

type WhenNotEquals<X, Y, A = X, B = never> = NotEquals<X, Y> extends true
  ? A
  : B
