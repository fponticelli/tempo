declare class PaperContext<Action> {
  constructor(
    canvas: HTMLCanvasElement,
    scope: PaperScope,
    project: Project,
    append: (item: Item) => void,
    dispatch: (action: Action) => void
  )
  readonly canvas: HTMLCanvasElement
  readonly scope: PaperScope
  readonly project: Project
  readonly append: (item: Item) => void
  readonly dispatch: (action: Action) => void
  mapAction<OtherAction>(
    f: (action: OtherAction) => Action
  ): PaperContext<OtherAction>
  conditionalMapAction<OtherAction>(
    f: (action: OtherAction) => Action | undefined
  ): PaperContext<OtherAction>
  withAppend(append: (item: Item) => void): PaperContext<Action>
  withDispatch<OtherAction>(
    dispatch: (action: OtherAction) => void
  ): PaperContext<OtherAction>
}
